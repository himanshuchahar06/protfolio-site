eldritch_mandala_shader.frag 
#ifdef GL_ES
precision mediump float;
#endif

uniform float u_time;
uniform vec2 u_resolution;
uniform float u_velocity; // Hand movement speed

void main() {
  vec2 uv = gl_FragCoord.xy / u_resolution.xy;
  vec2 center = uv - vec2(0.5);
  float r = length(center) * 2.0;
  float theta = atan(center.y, center.x);
  
  // Concentric runic oscillation bands
  float ring = sin(r * 40.0 - u_time * 8.0 * (1.0 + u_velocity));
  float spokes = sin(theta * 8.0 + u_time * 4.0);
  
  vec3 color = vec3(0.0, 0.95, 1.0) * smoothstep(0.05, 0.07, ring * spokes);
  gl_FragColor = vec4(color, 1.0);
}
